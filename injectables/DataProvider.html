import { Injectable } from '@angular/core';
import 'rxjs/add/operator/map';
import { SQLite, SQLiteObject } from '@ionic-native/sqlite';
import { Platform } from 'ionic-angular';
/*
  Generated class for the DataProvider provider.

  See https://angular.io/docs/ts/latest/guide/dependency-injection.html
  for more info on providers and Angular DI.
*/
@Injectable()
export class DataProvider {

  public database: SQLiteObject;

  constructor(public sqlite: SQLite, public platform:Platform) {
    platform.ready().then(()=>{
      console.log('Hello DataProvider Provider');

      // the DataProvider class constructor will 
      // 1. create a database called data.db if it does not exist
      // or
      // 2. if it already exist, then
      // 3. establish a connection to that database. 
      // 4. after that, call the function to create tables 
      this.sqlite.create({
        name: 'data.db',
        location: 'default'
      })
        .then((db: SQLiteObject) => {
          this.database = db;
          this.createTables();
        })
        .catch(e => console.log(`Error creating database: ${e}`));
    })
      

  }

  // create the table only if it does not exist. 
  // means 1st time it will create, then after that it will do nothing.
  // this function createTables() is only called from DataProvider constructor.
  
  async createTables() {
    try {
      await this.database.executeSql(
        `CREATE TABLE IF NOT EXISTS  scores (
          id INTEGER PRIMARY KEY,
          name TEXT,
          score INTEGER
          );` , []);
    } catch (e) {
      console.log(`Error creating tables: ${e}`);
      console.log("Error creating tables: " + e);
    }
  }

  // insert new record into table
  public insert(tableName, item) {
    let sqlText;
    let values;
    switch (tableName) {
      case "scores":
        sqlText = "INSERT INTO scores (name, score) VALUES (?,?)";
        values = [item.name || null, item.score || null]
        break;
      default:
        return;
    }
    return this.database.executeSql(sqlText, values);
    // executeSql means run this query (sqlText) on the database
  }

  // update existing record in the table
  public update(tableName, item) {
    let sqlText;
    let values;
    switch (tableName) {
      case "scores":
        sqlText = "UPDATE scores SET (name, score ) = ( ? , ? ) where id = ? ;";
        values = [item.name || null, item.score || null, item.id]
        break;
      default:
        return;

    }
    return this.database.executeSql(sqlText, values);
  }

  // delete record from table
  public delete(tableName, item) {
    let sqlText;
    let values;
    sqlText = `DELETE from ${tableName} where id = ? `;
    values = [item.id || null];
    return this.database.executeSql(sqlText, values);
  }

  // select all records from table
  public select(tableName) {
    let sqlText;
    let values = [];
    sqlText = `SELECT * from ${tableName} `;
    return this.database.executeSql(sqlText, values);
  }
}
