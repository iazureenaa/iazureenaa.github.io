import { Component } from '@angular/core';
import { NavController, ModalController } from 'ionic-angular';
import { FingerprintAIO } from '@ionic-native/fingerprint-aio';
import { PromiseObservable } from 'rxjs/observable/PromiseObservable';
import { DataProvider } from '../../providers/data/data';
import { ScoreListPage } from '../score-list/score-list';

import * as moment from 'moment';

@Component({
  selector: 'page-home',
  templateUrl: 'home.html',
  providers: [FingerprintAIO]
})
export class HomePage {
  authorised = null;
  imageSource = null;
  currentDate = "";


  constructor(public navCtrl: NavController, public fingerprint: FingerprintAIO, public dataProvider: DataProvider, public modalController: ModalController) {
    this.getBiometryAvailableType();
    this.updateTime();

  }

  public updateTime() {
    setInterval(function () {
      this.currentDate = moment().format('MMMM Do YYYY, h:mm:ss a');
    },
      1000);

  }

  public showFingerprintModal() {
    PromiseObservable.create(this.fingerprint.show({
      clientId: 'Fingerprint-Demo',
      clientSecret: 'password', //Only necessary for Android
      disableBackup: false,  //Only for Android(optional)
      localizedFallbackTitle: 'Use Pin', //Only for iOS
      localizedReason: 'Please authenticate' //Only for iOS

    }))
      .subscribe(
        value => {
          // alert("Success authenticate!:"+JSON.stringify(value));
          this.authorised = true;
          this.imageSource = "assets/img/tele-cherry-love.gif";
        },
        error => {
          // alert("Error authenticate!:"+JSON.stringify(error));
          this.authorised = false;
          this.imageSource = "assets/img/tele-cherry-sad.gif";
        });
  }
  public getBiometryAvailableType() {
    this.fingerprint.isAvailable().then(result => {
      alert("Result getBiometryAvailableType:" + result);
    }, error => {
      alert("Error getBiometryAvailableType:" + error);
    })

  }

  // openModal() {
  //   let modal = this.modalController.create(ScoreListPage);
  //   modal.present();
  // }

  scoreListPage() {
    this.navCtrl.push(ScoreListPage);
  }
}
