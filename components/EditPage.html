import { Component } from '@angular/core';
import { IonicPage, NavController, NavParams } from 'ionic-angular';
import { DataProvider } from '../../providers/data/data';

/**
 * This file lets you edit existing scores.
 */

@IonicPage()
@Component({
  selector: 'page-edit',
  templateUrl: 'edit.html',
})
export class EditPage {
  idFromScoreListPage = 0;
  nameFromScoreListPage = "";
  scoreFromScoreListPage = 0;

  constructor(public navCtrl: NavController, public navParams: NavParams, public dataProvider: DataProvider) {

    this.idFromScoreListPage = navParams.get("rowId");
    this.nameFromScoreListPage = navParams.get("rowName");
    this.scoreFromScoreListPage = navParams.get("rowScore");
  }

  editData(){
    var item = {name: this.nameFromScoreListPage, score: this.scoreFromScoreListPage, id: this.idFromScoreListPage}
    this.dataProvider.update("scores", item).then((data)=>{
      this.navCtrl.pop();
    })
  }

  ionViewDidLoad() {
    console.log('ionViewDidLoad EditPage');
  }

}
