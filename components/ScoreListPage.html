import { Component } from '@angular/core';
import { IonicPage, NavController, NavParams } from 'ionic-angular';
import { DataProvider } from '../../providers/data/data';
import { SQLiteObject } from '@ionic-native/sqlite';
import { AddDataPage } from '../add-data/add-data';
import { EditPage } from '../edit/edit';

/**
 * Generated class for the ScoreListPage page.
 *
 * See http://ionicframework.com/docs/components/#navigation for more info
 * on Ionic pages and navigation.
 */

@IonicPage()
@Component({
  selector: 'page-score-list',
  templateUrl: 'score-list.html',
})
export class ScoreListPage {

  public scoreList = [];
  // this.scoreList.push("izma")
  // scoreList = [ "izma" ]

  constructor(public navCtrl: NavController, public navParams: NavParams, public dataProvider: DataProvider) {
    // get data
    this.getData();
  }

  // to delete the item by sliding and press delete icon
  deleteData(rowId) {
    var item = { id: rowId };
    this.dataProvider.delete("scores", item).then((data) => {
      // refresh data
      this.getData();

    })
  }

  // read from database scores table and push the result into this.scoreList variable
  getData() {
    this.dataProvider.select("scores").then((data) => {
      // make scoreList = empty array first before pushing data from "select" result. 
      // otherwise it will keep adding to existing data which causes front-end to show repeated rows.
      this.scoreList = [];
      for (var i = 0; i < data.rows.length; i++) {
        this.scoreList.push(
          {
            id: data.rows.item(i).id,
            name: data.rows.item(i).name,
            score: data.rows.item(i).score
          }
        )
      }
      // scoreList = [ { "rowid" : 0, "name" : "Izma",  score: 1000 } , { "rowid" : 1, "name" : "Izma",  score: 1000 } , { } , .... ]
    });
  }

  addData(){
    this.navCtrl.push(AddDataPage);
  }

  editData(id,name,score){
    this.navCtrl.push(EditPage, { rowId: id, rowName: name, rowScore: score });
  }


  ionViewDidLoad() {
    console.log('ionViewDidLoad ScoreListPage');
  }

  // 
  ionViewDidEnter(){
    this.getData();
  }

}
